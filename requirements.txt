# DocFlow — Dependency Reference
# ─────────────────────────────────────────────────────────────────────────────
# This is a Node.js project. Dependencies are managed with npm, NOT pip.
# Install everything by running the commands in the SETUP section below.
#
# ─────────────────────────────────────────────────────────────────────────────
# PREREQUISITES
# ─────────────────────────────────────────────────────────────────────────────
#   Node.js  >= 18.x   https://nodejs.org/
#   npm      >= 9.x    (bundled with Node.js)
#
# ─────────────────────────────────────────────────────────────────────────────
# SETUP (run in order)
# ─────────────────────────────────────────────────────────────────────────────
#   1. Clone / unzip the project
#   2. cd server && npm install
#   3. cp server/.env.example server/.env   (then fill in your keys)
#   4. cd ../client && npm install
#   5. Start the server:  cd ../server && npm run dev   (port 3001)
#   6. Start the client:  cd ../client && npm run dev   (port 5173)
#
# ─────────────────────────────────────────────────────────────────────────────
# SERVER  (server/package.json)
# ─────────────────────────────────────────────────────────────────────────────
# Runtime dependencies:
#   @google/generative-ai   ^0.21.0   — Gemini AI for per-page field extraction
#   @supabase/supabase-js   ^2.39.0   — Supabase client (DB + storage)
#   cors                    ^2.8.5    — Cross-origin request headers
#   dotenv                  ^16.3.1   — Load .env into process.env
#   express                 ^4.18.2   — HTTP server / routing
#   multer                  ^1.4.5    — Multipart form-data (PDF upload)
#   pdf-lib                 ^1.17.1   — PDF parsing (AcroForm + content streams)
#
# Dev dependencies:
#   nodemon                 ^3.0.2    — Auto-restart server on file changes
#

# SUPABASE TABLE  (run once in Supabase SQL editor)
# ─────────────────────────────────────────────────────────────────────────────
#   CREATE TABLE parsed_documents (

#     fields      jsonb DEFAULT '[]',
#     checkboxes  jsonb DEFAULT '[]',
#     created_at  timestamptz DEFAULT now()
#   );
#
#   -- Enable public read (adjust RLS as needed for production)
#   ALTER TABLE parsed_documents ENABLE ROW LEVEL SECURITY;
